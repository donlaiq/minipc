def make_graph(adj_list):
    if not adj_list: return None
    nodes = [Node(i+1) for _ in range(len(adj_list))]
    for i, neighbors in enumerate(adj_list):
        for n in neighbors:
            nodes[i].neighbors.append(nodes[n-1])
    return nodes[0]

def count_nodes(node):
    if not node: return 0
    visited = set()
    stack = [node]
    while stack:
        n = stack.pop()
        if n.val in visited: continue
        visited.add(n.val)
        for nb in n.neighbors:
            if nb.val not in visited:
                stack.append(nb)
    return len(visited)
