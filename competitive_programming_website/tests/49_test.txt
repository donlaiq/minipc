def make_list(arr):
    if not arr: return None
    head = ListNode(arr[0])
    curr = head
    for v in arr[1:]:
        curr.next = ListNode(v)
        curr = curr.next
    return head

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next
