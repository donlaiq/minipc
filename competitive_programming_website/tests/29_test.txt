def test_queue(operations):
    q = MyQueue()
    results = []
    for op in operations:
        if op["op"] == "push":
            q.push(op["x"])
            results.append(None)
        elif op["op"] == "pop":
            results.append(q.pop())
        elif op["op"] == "peek":
            results.append(q.peek())
        elif op["op"] == "empty":
            results.append(q.empty())
    return results
