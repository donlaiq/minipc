def make_list(arr):
    if not arr: return None
    head = ListNode(arr[0])
    curr = head
    for v in arr[1:]:
        curr.next = ListNode(v)
        curr = curr.next
    return head

def isPalindromeWrapper(arr):
    head = make_list(arr)
    return isPalindrome(head)
