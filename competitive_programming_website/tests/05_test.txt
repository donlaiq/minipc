# Helper to create a linked list from a Python list
def make_list(arr):
    if not arr: return None
    head = ListNode(arr[0])
    curr = head
    for v in arr[1:]:
        curr.next = ListNode(v)
        curr = curr.next
    return head
def list_to_array(head):
    result = []
    while head:
        result.append(head.val)
        head = head.next
    return result
def mergeKListsWrapper(lists):
    head = mergeKLists([make_list(arr) if isinstance(arr, list) else arr for arr in lists])
    return list_to_array(head)

# Test the wrapper via the solution function
def test_mergeKLists(input_lists):
    lists = [make_list(arr) if arr is not None else None for arr in input_lists]
    result_head = mergeKLists(lists)
