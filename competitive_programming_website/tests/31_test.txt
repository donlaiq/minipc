def make_bst(arr):
    if not arr: return None
    root = TreeNode(arr[0])
    for v in arr[1:]:
        node = root
        while True:
            if v < node.val:
                if not node.left: node.left = TreeNode(v); break
                else: node = node.left
            else:
                if not node.right: node.right = TreeNode(v); break
                else: node = node.right
    return root

def find_p_q(root, p_val, q_val):
    return lowestCommonAncestor(root, TreeNode(p_val), TreeNode(q_val))

def lcaWrapper(arr, p_val, q_val):
    root = make_bst(arr)
    node = find_p_q(root, p_val, q_val)
    return node.val if node else None
